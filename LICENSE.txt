CREATE TABLE "users_info" (
	"_id" serial NOT NULL,
	"username" varchar(255) UNIQUE,
	"password" varchar(255),
	"email" varchar(255) UNIQUE,
	"interests" varchar(255),
	CONSTRAINT users_pk PRIMARY KEY ("_id")
) WITH (
  OIDS=FALSE
);



CREATE TABLE "interests_selections" (
	"_id" serial NOT NULL,
	"Interest" serial(50) UNIQUE,
	CONSTRAINT interests_pk PRIMARY KEY ("_id")
) WITH (
  OIDS=FALSE
);



CREATE TABLE "join-users-to-interests" (
	"_id" serial NOT NULL,
	"user_id" varchar UNIQUE,
	"increment_id" varchar,
	CONSTRAINT join-users-to-interests_pk PRIMARY KEY ("_id")
) WITH (
  OIDS=FALSE
);



ALTER TABLE "users" ADD CONSTRAINT "users_fk0" FOREIGN KEY ("interests") REFERENCES "interests"("_id");


ALTER TABLE "join-users-to-interests" ADD CONSTRAINT "join-users-to-interests_fk0" FOREIGN KEY ("user_id") REFERENCES "users"("_id");
ALTER TABLE "join-users-to-interests" ADD CONSTRAINT "join-users-to-interests_fk1" FOREIGN KEY ("increment_id") REFERENCES "interests"("_id");
